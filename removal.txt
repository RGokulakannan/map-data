def removeAllFiles(folder_loc):
    files = glob.glob(os.path.join(folder_loc, '*'))
    for file in files:
        try: 
            os.remove(file)
        except Exception as e:
            print(f"Error deleting {file}: {e}")

def split_df(max_cost, max_qty, max_row):
    max_cost = int(max_cost)
    max_qty = int(max_qty)
    max_row = int(max_row)

    df = pd.read_excel("input_split.xlsx")

    products = df["FNSKU"].to_numpy()
    costs = df["Cost per Unit"].to_numpy()
    qtys = df["Quantity"].to_numpy()

    splits = []
    start = len(df)

    current_cost = 0
    current_qty = 0

    # Loop backwards
    for i in range(len(df) - 1, -1, -1):
        q = qtys[i]
        c = costs[i] * q

        # Remove invalid rows
        if q > max_qty:
            print(products[i], f"in row number {i + 2} has higher quantity")
            df.drop(index=i, inplace=True)
            continue

        if c > max_cost:
            print(products[i], f"in row number {i + 2} has higher cost price")
            df.drop(index=i, inplace=True)
            continue

        # Check limits
        if (
            start - i > max_row
            or current_cost + c > max_cost
            or current_qty + q > max_qty
        ):
            splits.append(df.iloc[i + 1:start])
            start = i + 1

            current_cost = 0
            current_qty = 0

        current_cost += c
        current_qty += q

    # Add remaining rows
    splits.append(df.iloc[0:start])

    # Save outputs
    for idx, split_df in enumerate(reversed(splits), start=1):
        if not split_df.empty:
            split_df.to_excel(f"data\\Output_{idx}.xlsx", index=False)

def method_four():
    disable_all_buttons()
    t1 = time.time()    
    def consolidate_excel_files(source_folder, output_file):
        all_dataframes = []

        for file in os.listdir(source_folder):
            if file.endswith(".xlsx") or file.endswith(".xls"):
                file_path = os.path.join(source_folder, file)
                try:
                    df = pd.read_excel(file_path)
                    df['SourceFile'] = file  # Optional: Add column to track source file
                    all_dataframes.append(df)
                    print(f"Loaded: {file}")
                except Exception as e:
                    print(f"Failed to read {file}: {e}")

        if all_dataframes:
            combined_df = pd.concat(all_dataframes, ignore_index=True)
            combined_df.to_excel(output_file, index=False)
            print(f"\n✅ Consolidated Excel saved to: {output_file}")
        else:
            print("⚠️ No Excel files found or loaded.")
    # Example usage
    source_folder = DOWNLOADDIR
    output_excel = "consolidated.xlsx"
    consolidate_excel_files(source_folder, output_excel)
    t2 = time.time()
    print("Completed in",round((t2-t1)/60, 2), "minutes")
    enable_all_buttons()
