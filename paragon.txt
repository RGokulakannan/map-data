def cookie_management():

    chrome_options.add_argument("--start-maximized")

    chrome_options.add_experimental_option('excludeSwitches', ['enable-logging'])
    driver = webdriver.Chrome(service=s, options=chrome_options)
    return driver

def scip(vc):
    d = []
    primary = False
    to_email_type = ''
    # primary_email = []
    cc_email_list = []
    to_email = ''

    # if 1 == 1:
    try:
        cnt_list = ['Primary Account Rep', 'Direct Buying', 'Supply Chain Operations', 'Warehouse']
        browser.get(f'https://scip.corp.amazon.com/contacts/vendorcode/{str(vc)}')
        WebDriverWait(browser, 60).until(EC.presence_of_element_located((By.XPATH,'/html/body/div/div[2]/kat-tabs/kat-tab[1]/div[2]/kat-data-table/table/tbody')))
        time.sleep(5)
        src = browser.page_source
        soup = BeautifulSoup(browser.page_source, 'lxml')
        table = soup.find("table", {'aria-labelledby': 'katal-id-0'})
        table_body = table.find("tbody")
        trs = table_body.find_all("tr")

        if "Primary Account Rep" in str(trs):
            primary = True
            to_email_type = 'Primary Account Rep'
        elif "Direct Buying" in str(trs):
            to_email_type = 'Direct Buying'
        elif "Supply Chain Operations" in str(trs):
            to_email_type = "Supply Chain Operations"
        elif "Warehouse" in str(trs):
            to_email_type = "Warehouse"

        for tr in trs:
            if any(cnt in str(tr) for cnt in cnt_list):
                tds = tr.find_all("td")
                index = 1
                for td in tds:
                    if index == 1:
                        type = str(td).partition("<h5>")[2].partition("</h5>")[0]
                    if index == 3:
                        if type == to_email_type and to_email == '':
                            to_email = str(td).partition("<br/>")[2].partition("</td>")[0]
                            if "@amazon" in to_email : to_email = ""
                        else:
                            em = str(td).partition("<br/>")[2].partition("</td>")[0]
                            if em not in cc_email_list:
                                if "@amazon" not in em : cc_email_list.extend([em])
                    index = index + 1

    except Exception:
        print("Exception occurred while fetching contacts from SCIP")
        return "Exception occurred while fetching contacts from SCIP"

    print("contacts=", d)
    d = cc_email_list

    if to_email == '':
        return None, None
    elif len(cc_email_list) > 0:
        if to_email in d:
            d.remove(to_email)
            if len(d) == 0:
                d = None
    else:
        d = None

    return to_email, d
